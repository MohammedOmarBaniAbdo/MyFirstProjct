int Roundem(int form, int to) {
	int Roun = rand() % (to - form + 1) + form;
	return Roun;
}
bool Check() {
	char Ch;
	cout << "Plet Agen? (Y/N)\n";
	cin >> Ch;
	if (Ch == 'Y' || Ch == 'y') {
		return true;
	}
	else if (Ch == 'N' || Ch == 'n') {
		return false;
	}
	else {
		cout << "Enter Right Value\n";
		return Check();
	}
}
